        function changeLanguage(lang) {
            currentLanguage = lang;
            document.querySelectorAll('.lang-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            document.querySelector(`[data-lang="${lang}"]`).classList.add('active');
            
            document.querySelectorAll('[data-translate]').forEach(el => {
                const key = el.getAttribute('data-translate');
                if (translations[lang][key]) {
                    el.textContent = translations[lang][key];
                }
            });
            
            if (document.getElementById('main-menu').style.display !== 'none') {
                renderMainMenu();
            } else {
                const currentCategory = document.getElementById('category-title').getAttribute('data-category');
                if (currentCategory) {
                    showCategory(currentCategory);
                }
            }
        }

        function renderMainMenu() {
            const mainMenu = document.getElementById('main-menu');
            mainMenu.innerHTML = '';
            
            const categories = Object.keys(menuData);
            categories.forEach(category => {
                const card = document.createElement('div');
                card.className = 'category-card';
                card.onclick = () => showCategory(category);
                
                const categoryName = translations[currentLanguage][category] || category;
                const itemCount = menuData[category].length;
                const imageSrc = getImageForCategory(category);
                const itemsText = itemCount === 1 ? translations[currentLanguage].item : translations[currentLanguage].items;
                
                card.innerHTML = `
                    <img src="${imageSrc}" alt="${categoryName}" class="category-image" 
                         onclick="event.stopPropagation(); openImageModal('${imageSrc}')"
                         onerror="this.src=''; this.style.display='none';">
                    <div class="category-content">
                        <h3 style="font-size: 26px; font-weight: bold; color: #d4af37; margin-bottom: 8px; text-align: center; letter-spacing: 1px;">${categoryName}</h3>
                        <p style="color: #999; font-size: 15px; text-align: center;">${itemCount} ${itemsText}</p>
                    </div>
                `;
                
                mainMenu.appendChild(card);
            });
        }

        function showCategory(category) {
            document.getElementById('main-menu').style.display = 'none';
            document.getElementById('category-details').style.display = 'block';
            
            const categoryTitle = document.getElementById('category-title');
            categoryTitle.textContent = translations[currentLanguage][category] || category;
            categoryTitle.setAttribute('data-category', category);
            
            const itemsContainer = document.getElementById('items-container');
            itemsContainer.innerHTML = '';
            
            menuData[category].forEach(item => {
                const itemCard = document.createElement('div');
                itemCard.className = 'item-card';
                const imageSrc = getImageForItem(item.name, category);
                const translatedName = translations[currentLanguage][item.name] || item.name;
                
                itemCard.innerHTML = `
                    <div class="item-image-container" onclick="event.stopPropagation(); openImageModal('${imageSrc}')">
                        <img src="${imageSrc}" alt="${translatedName}" class="item-image"
                             onerror="this.src=''; this.style.display='none';">
                    </div>
                    <div class="item-content">
                        <div>
                            <h4 style="font-size: 22px; font-weight: bold; color: #333; margin-bottom: 5px;">${translatedName}</h4>
                        </div>
                        <div class="price-tag">$${item.price}</div>
                    </div>
                `;
                
                itemsContainer.appendChild(itemCard);
            });
        }

        function showMainMenu() {
            document.getElementById('main-menu').style.display = 'grid';
            document.getElementById('category-details').style.display = 'none';
        }
